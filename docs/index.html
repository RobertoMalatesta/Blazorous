<h1>Blazorous</h1>
<h2>Maintainable CSS with Blazor</h2>
<Dynamic TagName="a" css="@link" css="@primary" href="https://github.com/chanan/Blazorous">Github</Dynamic>
<Dynamic TagName="a" css="@link" css="@secondary" href="/glamorexamples">Glamor Examples</Dynamic>
</Dynamic>

@functions {
Css center = Css.CreateNew().AddRule("text-align", "center").AddRule("margin-bottom", "30px");
Css link = Css.CreateNew().AddRule("border", "1px solid #F67982")
                          .AddRule("width", "11em")
                          .AddRule("padding", "0.7em 0")
                          .AddRule("text-decoration", "none")
                          .AddRule("border-radius", 4)
                          .AddRule("display", "inline-block")
                          .AddRule("margin", ".5em 1em")
                          .AddRule("transition", "all .3s");
Css primary = Css.CreateNew().AddRule("background-color", "#CC3A4B")
                             .AddRule("color", "#fff")
                             .OpenSelector(":hover")
                                .AddRule("background-color", "rgba(255, 255, 255, 0.5)")
                                .AddRule("color", "#DA233C")
                             .CloseSelector();
Css secondary = Css.CreateNew().AddRule("background-color", "rgba(255, 255, 255, 0.5)")
                               .AddRule("color", "#DA233C")
                               .OpenSelector(":hover")
                                    .AddRule("background-color", "#CC3A4B")
                                    .AddRule("color", "#fff")
                               .CloseSelector();
}
